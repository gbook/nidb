NiDB API — Client / Server Flow

sequenceDiagram
    autonumber
    participant C  as Client
    participant AP as api.php
    participant DB as MySQL DB

    rect rgb(235, 245, 255)
        Note over C,AP: Every request — pre-flight checks
        C->>AP: POST /api.php  { action, ... }
        AP->>AP: Reject if not HTTPS
        AP->>AP: Reject if not POST
        AP->>AP: Reject if action not in allowlist
    end

    alt action = Authenticate
        rect rgb(230, 255, 235)
            Note over C,DB: Authentication & session creation
            C->>AP: user, token
            AP->>AP: Reject if user or token missing
            AP->>DB: SELECT apiuser_id, credential, is_active
FROM api_users WHERE username = ?
            DB-->>AP: user row  (or not found)
            AP->>AP: password_verify(token, hash)
always runs — prevents timing attacks
            AP-->>C: 401 Authentication failed
(bad credentials or inactive user)
            AP->>DB: DELETE expired sessions for this user
(idle > 10 min — lazy cleanup)
            AP->>DB: COUNT active sessions for this user
            DB-->>AP: active session count
            AP-->>C: 429 Max concurrent sessions reached
(if count ≥ 2)
            AP->>AP: generateSessionId()
128 random alphanumeric chars
            AP->>DB: INSERT api_sessions
( SHA-256(sessionID), ip_address, timestamps )
            AP->>DB: UPDATE api_users
SET login_count+1, last_access=NOW()
            AP-->>C: 200 { sessionID: "<128-char string>" }
        end

    else action requires sessionID
        rect rgb(255, 248, 225)
            Note over C,DB: Session validation — runs for all remaining actions
            C->>AP: sessionID  (+ action-specific params)
            AP->>AP: Reject if sessionID missing or
not exactly 128 alphanumeric chars
            AP->>DB: SELECT session WHERE
session_hash = SHA-256(sessionID)
AND is_active = 1
AND last_access > NOW() - 10 min
            DB-->>AP: session row  (or not found)
            AP-->>C: 401 Invalid or expired session
            AP->>DB: UPDATE api_sessions
SET last_access = NOW()
        end

        alt action = StartTransaction
            rect rgb(245, 235, 255)
                AP->>DB: Create transaction record
                DB-->>AP: transactionID
                AP-->>C: 200 { transactionID: "..." }
            end

        else action = ImportObject
            rect rgb(245, 235, 255)
                C->>AP: squirrelObject  [FILE]
                AP->>AP: Reject if extension ≠ .sqrl
                AP->>AP: Reject if size > 512 MB
                AP->>AP: Reject if MIME type not in allowlist
                AP->>DB: Process .sqrl file contents
                DB-->>AP: per-object results
                AP-->>C: 200 { results: [ { ... }, ... ] }
            end

        else action = EndTransaction
            rect rgb(245, 235, 255)
                AP->>DB: Commit / rollback transaction
                DB-->>AP: ok
                AP-->>C: 200 Transaction ended
            end

        else action = GetProjectList
            rect rgb(245, 235, 255)
                AP->>DB: SELECT projectID, projectName
for this user
                DB-->>AP: project rows
                AP-->>C: 200 { projects: [ { ... }, ... ] }
            end

        else action = GetAcceptedObjects
            rect rgb(245, 235, 255)
                AP->>DB: SELECT accepted object types
                DB-->>AP: object type list
                AP-->>C: 200 { objectTypes: [ ... ] }
            end
        end
    end
